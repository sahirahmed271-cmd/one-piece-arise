<script>
let board=document.getElementById('board'),message=document.getElementById('message'),cells=[],turn='X';
for(let i=0;i<9;i++){let c=document.createElement('div');c.className='cell';c.addEventListener('click',()=>makeMove(c,i));board.appendChild(c);cells.push(c);}
let winCombos=[[0,1,2],[3,4,5],[6,7,8],[0,3,6],[1,4,7],[2,5,8],[0,4,8],[2,4,6]];
function makeMove(c,i){if(c.textContent||message.textContent)return;c.textContent=turn;checkWin();turn=turn==='X'?'O':'X';}
function checkWin(){for(let combo of winCombos){if(cells[combo[0]].textContent&&cells[combo[0]].textContent===cells[combo[1]].textContent&&cells[combo[0]].textContent===cells[combo[2]].textContent){message.textContent=cells[combo[0]].textContent+' Wins!';return;}}
if(cells.every(c=>c.textContent)){message.textContent='Draw!';}}
